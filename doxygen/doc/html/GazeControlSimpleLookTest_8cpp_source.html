70 /***********************************************************************************/
71 void GazeControlSimpleLookTest::run()
72 {
73  IGazeControl *igaze;
74  RTF_TEST_CHECK(driver.view(igaze),"Opening the view on the device!");
75 
76  bool done;
77 
78  Vector fp;
79  double t0=Time::now();
80  while (Time::now()-t0<5.0)
81  {
82  done=igaze->getFixationPoint(fp);
83  if (done)
84  break;
85  Time::delay(0.1);
86  }
87  RTF_TEST_CHECK(done,"Initial fixation-point retrieved!");
88 
89  RTF_TEST_REPORT("Setting up the context");
90  int context;
91  igaze->storeContext(&context);
92 
93  RTF_TEST_REPORT("Looking at the target");
94  Vector fpd(3,0.0); fp[0]=-0.4;
95  igaze->lookAtFixationPoint(fpd);
96 
97  RTF_TEST_REPORT("Waiting");
98  igaze->waitMotionDone(1.0,5.0);
99 
100  igaze->checkMotionDone(&done);
101  RTF_TEST_CHECK(done,"Target reached!");
102 
103  RTF_TEST_REPORT("Going back");
104  igaze->lookAtFixationPoint(fp);
105 
106  RTF_TEST_REPORT("Waiting");
107  igaze->waitMotionDone(1.0,5.0);
108 
109  igaze->checkMotionDone(&done);
110  RTF_TEST_CHECK(done,"Done!");
111 
112  RTF_TEST_REPORT("Cleaning up the context");
113  igaze->restoreContext(context);
114  igaze->deleteContext(context);
115 }
